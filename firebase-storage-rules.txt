rules_version = '2';

service firebase.storage {
  match /b/{bucket}/o {
    // Profile photos: authenticated users can read any profile photo
    match /profilePhotos/{allFiles=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }
    
    // All other files: authenticated users can read/write
    match /{allPaths=**} {
      allow read, write: if request.auth != null;
    }
  }
}
